ol.animation


function fly(map, frameState) {
  if (frameState.time < start) {
    frameState.animate = true;
    frameState.viewHints[ol.ViewHint.ANIMATING] += 1;
    return true;
  } else if (frameState.time < start + duration) {
    var delta = easing((frameState.time - start) / duration);
    var deltaResolution = resolution - frameState.view2DState.resolution;
    frameState.animate = true;
    frameState.view2DState.resolution += delta * deltaResolution;
    frameState.viewHints[ol.ViewHint.ANIMATING] += 1;
    return true;
  } else {
    return false;
  }
};

ol.interaction / ol.control

  • ol.interaction
    • No DOM presence
    • Handles user input
    • e.g. DragPan, DoubleClickZoom, TouchRotate, etc.
  • ol.control
    • DOM presence
    • Drive and are driven by the map
    • e.g. Zoom buttons, Scale line, Attribution

ol.interaction


var map = new ol.Map({
  interactions: ol.interaction.defaults({}, [
    new ol.interaction.DragRotateAndZoom()
  ])
  // ...
});
